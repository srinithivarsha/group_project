@app.route("/process_purchase", methods=["POST"]) def process_purchase(): """Old purchase route - handles direct purchase without confirmation""" if 'user_id' not in session: return redirect(url_for('login')) card_number = session['card_number'] db = get_db() cur = db.cursor() # Dictionary to track quantities by item name items_purchased = [] for key in request.form: if key.startswith('quantity_'): item_name = key.replace('quantity_', '') qty_str = request.form.get(key, '0').strip() # Handle both integer and decimal quantities try: qty = float(qty_str) if '.' in qty_str else int(qty_str) except (ValueError, TypeError): qty = 0 if qty > 0: price_str = request.form.get(f'price_{item_name}', '0') try: price = float(price_str) except (ValueError, TypeError): price = 0.0 items_purchased.append({ 'name': item_name, 'quantity': qty, 'price': price }) # Save each item to database for item in items_purchased: cur.execute(""" INSERT INTO purchases (card_number, item_name, quantity, price, purchase_date) VALUES (?, ?, ?, ?, datetime('now')) """, (card_number, item['name'], item['quantity'], item['price'])) db.commit() db.close() if items_purchased: flash(f"Purchase saved successfully! {len(items_purchased)} items purchased.", "success") else: flash("No items selected for purchase.", "warning") return redirect(url_for('history'))
